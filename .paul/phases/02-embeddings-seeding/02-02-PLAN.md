<objective>
## Goal
Create a NestJS `nest-commander` seed command (`pnpm seed`) that populates NeonDB with the full POC fixture — 2 venues, 5 suppliers, 7 stock categories, ~25 stock items, 6 SOP documents, venue contacts — enriches each SOP with a Claude-generated `aiSummary` + `aiTags`, embeds every stock item and SOP via the existing `EmbeddingsService`, and writes vectors to the `vector(1024)` columns via `$executeRaw`.

## Purpose
Phase 3 (retrieval) and Phase 4 (chat) need seeded, embedded data to exercise. Running the seeder transforms an empty schema into a working POC dataset in one command. The enrichment step (Claude → summary + tags) is domain-critical: search on SOPs works by embedding `title + aiSummary + tags + content` (see PAUL.md §7.1), not raw content alone — so the seeder is where that AI-generated metadata lands.

    **3. Create `apps/api/src/modules/seed/enrichment.service.ts`:**
    ```ts
    import { Injectable, Logger, OnModuleInit } from '@nestjs/common'
    import Anthropic from '@anthropic-ai/sdk'

    export type SopEnrichment = { summary: string; tags: string[] }

    @Injectable()
    export class EnrichmentService implements OnModuleInit {
      private readonly logger = new Logger(EnrichmentService.name)
      private client!: Anthropic

      onModuleInit() {
        const apiKey = process.env.ANTHROPIC_API_KEY
        if (!apiKey) throw new Error('ANTHROPIC_API_KEY is not set — add it to .env at repo root')
        this.client = new Anthropic({ apiKey })
      }

      async enrichSop(doc: { title: string; category: string; content: string }): Promise<SopEnrichment | null> {
        const prompt = `You are enriching an SOP document for a hospitality operations AI system.

Given the following SOP document, return a JSON object with:
- "summary": A 1-2 sentence plain-English summary of what this document covers
- "tags": An array of 5-10 lowercase semantic tags (e.g. ["ice_machine", "equipment", "error_codes", "troubleshooting"])

Respond ONLY with valid JSON. No markdown, no explanation.

Title: ${doc.title}
Category: ${doc.category}
Content:
${doc.content}`

        const response = await this.client.messages.create({
          model: 'claude-sonnet-4-20250514',
          max_tokens: 1024,
          messages: [{ role: 'user', content: prompt }],
        })

        const text = response.content
          .filter((b) => b.type === 'text')
          .map((b) => (b as { text: string }).text)
          .join('')

        try {
          const parsed = JSON.parse(text) as SopEnrichment
          if (typeof parsed.summary !== 'string' || !Array.isArray(parsed.tags)) {
            this.logger.warn(`Enrichment for "${doc.title}" returned wrong shape; skipping`)
            return null
          }
          return parsed
        } catch {
          this.logger.warn(`Enrichment for "${doc.title}" returned invalid JSON; skipping`)
          return null
        }
      }
    }
    ```
    Per PAUL.md §7.3: log and skip on invalid JSON, do NOT crash the seeder. The Prompt is copied verbatim from §7.3.

    Model hardcoded to `claude-sonnet-4-20250514` per PROJECT.md. `max_tokens: 1024` is ample for a short JSON object.

    **1. Create `apps/api/src/modules/seed/seed.command.ts`:**
    ```ts
    import { Command, CommandRunner } from 'nest-commander'
    import { Logger } from '@nestjs/common'
    import { prisma } from '@gm-ai/database'
    import { EmbeddingsService } from '../embeddings/embeddings.service'
    import { EnrichmentService } from './enrichment.service'
    import { venues, suppliers, stockCategories, stockItems, sopDocuments, venueContacts } from './seed-data'

    @Command({ name: 'seed', description: 'Seed NeonDB with POC fixture + embeddings' })
    export class SeedCommand extends CommandRunner {
      private readonly logger = new Logger(SeedCommand.name)

      constructor(
        private readonly embeddings: EmbeddingsService,
        private readonly enrichment: EnrichmentService,
      ) { super() }

      async run(): Promise<void> {
        this.logger.log('Wiping existing seed data...')
        // Delete in reverse dependency order. Using deleteMany on the 9 tables
        // (ChatMessage + ChatConversation first — future-proof even if empty).
        await prisma.chatMessage.deleteMany()
        await prisma.chatConversation.deleteMany()
        await prisma.purchaseOrderItem.deleteMany()
        await prisma.purchaseOrder.deleteMany()
        await prisma.venueContact.deleteMany()
        await prisma.sopDocument.deleteMany()
        await prisma.stockItem.deleteMany()
        await prisma.stockCategory.deleteMany()
        await prisma.supplier.deleteMany()
        await prisma.venue.deleteMany()

        this.logger.log(`Inserting ${venues.length} venues...`)
        await prisma.venue.createMany({ data: venues as unknown as typeof venues[number][] })

        this.logger.log(`Inserting ${suppliers.length} suppliers...`)
        await prisma.supplier.createMany({ data: suppliers as unknown as typeof suppliers[number][] })

        this.logger.log(`Inserting ${stockCategories.length} stock categories...`)
        await prisma.stockCategory.createMany({ data: stockCategories as unknown as typeof stockCategories[number][] })

        this.logger.log(`Inserting ${stockItems.length} stock items + embedding...`)
        const stockEmbeddingTexts = stockItems.map((s) => {
          const catName = stockCategories.find((c) => c.id === s.categoryId)?.name ?? ''
          return `${s.name}. Category: ${catName}. Unit: ${s.unitSize ?? s.unit}. ${s.notes ?? ''}`.trim()
        })
        const stockVectors = await this.embeddings.embedDocuments(stockEmbeddingTexts)
        // Insert row, then write vector via $executeRaw
        for (let i = 0; i < stockItems.length; i++) {
          const item = stockItems[i]
          const vec = stockVectors[i]
          await prisma.stockItem.create({
            data: { ...item, embeddingText: stockEmbeddingTexts[i] },
          })
          await prisma.$executeRawUnsafe(
            `UPDATE "StockItem" SET embedding = $1::vector WHERE id = $2`,
            `[${vec.join(',')}]`,
            item.id ?? '',  // must match created id — use deterministic IDs in seed-data.ts
          )
        }
        // NOTE: createMany returns a count, not rows. Deterministic IDs from
        // seed-data.ts flow through to stock_items.id so the UPDATE matches.

        this.logger.log(`Enriching + embedding ${sopDocuments.length} SOPs...`)
        for (const doc of sopDocuments) {
          const enriched = await this.enrichment.enrichSop(doc)
          const aiSummary = enriched?.summary ?? null
          const aiTags = enriched?.tags ?? []
          const embeddingText = `${doc.title}. ${aiSummary ?? ''}. Tags: ${aiTags.join(', ')}. ${doc.content}`
          const [vec] = await this.embeddings.embedDocuments([embeddingText])
          const sop = await prisma.sopDocument.create({
            data: { ...doc, aiSummary, aiTags },
          })
          await prisma.$executeRawUnsafe(
            `UPDATE "SopDocument" SET embedding = $1::vector WHERE id = $2`,
            `[${vec.join(',')}]`,
            sop.id,
          )
        }

        this.logger.log(`Inserting ${venueContacts.length} venue contacts...`)
        await prisma.venueContact.createMany({ data: venueContacts as unknown as typeof venueContacts[number][] })

        this.logger.log('Seed complete.')
      }
    }
    ```

    Important implementation notes:
    - Use `$executeRawUnsafe` (not `$executeRaw`) because the vector value must be pre-built as a string with commas (Prisma's `$executeRaw` template-tag escapes commas). The placeholder-based call is safe because we control both inputs (not user-provided).
    - Deterministic IDs in seed-data.ts (`id: '00000000-...-0001'`-style or real uuidv4s hand-generated) — critical so the `UPDATE ... WHERE id = $2` hits the right row without a round-trip select.
    - Stock embeddings are batched (one Voyage call); SOP embeddings are per-doc because each needs Claude enrichment first.
    - Delete-all-first is the idempotency strategy (AC-4). For POC this is simpler than upserts and avoids stale-embedding issues.

    ```ts
    import { config } from 'dotenv'
    import { resolve } from 'node:path'
    config({ path: resolve(__dirname, '../../../.env') })

    import { prisma } from '@gm-ai/database'

    async function main() {
      const [venueN, supplierN, catN, stockN, sopN, contactN] = await Promise.all([
        prisma.venue.count(),
        prisma.supplier.count(),
        prisma.stockCategory.count(),
        prisma.stockItem.count(),
        prisma.sopDocument.count(),
        prisma.venueContact.count(),
      ])

      // Embedding coverage — vector columns; Prisma can't count on Unsupported,
      // so use raw SQL.
      const [stockEmbedded, sopEmbedded, sopSummarised, sopTagged] = await Promise.all([
        prisma.$queryRawUnsafe<{ n: bigint }[]>(`SELECT COUNT(*)::bigint AS n FROM "StockItem" WHERE embedding IS NOT NULL`),
        prisma.$queryRawUnsafe<{ n: bigint }[]>(`SELECT COUNT(*)::bigint AS n FROM "SopDocument" WHERE embedding IS NOT NULL`),
        prisma.$queryRawUnsafe<{ n: bigint }[]>(`SELECT COUNT(*)::bigint AS n FROM "SopDocument" WHERE "aiSummary" IS NOT NULL AND length("aiSummary") > 0`),
        prisma.$queryRawUnsafe<{ n: bigint }[]>(`SELECT COUNT(*)::bigint AS n FROM "SopDocument" WHERE array_length("aiTags", 1) >= 1`),
      ])

      const checks: ReadonlyArray<readonly [string, boolean]> = [
        ['venues = 2', venueN === 2],
        ['suppliers = 5', supplierN === 5],
        ['stock_categories = 7', catN === 7],
        ['stock_items >= 25', stockN >= 25],
        ['sop_documents = 6', sopN === 6],
        ['venue_contacts >= 3', contactN >= 3],
        ['every stock_item has embedding', Number(stockEmbedded[0].n) === stockN],
        ['every sop_document has embedding', Number(sopEmbedded[0].n) === sopN],
        ['every sop_document has aiSummary', Number(sopSummarised[0].n) === sopN],
        ['every sop_document has aiTags', Number(sopTagged[0].n) === sopN],
      ]

      let failed = 0
      for (const [name, ok] of checks) {
        console.log(ok ? `\u2713 ${name}` : `\u2717 ${name}`)
        if (!ok) failed++
      }
      console.log(`\nrow counts: venues=${venueN} suppliers=${supplierN} cats=${catN} stock=${stockN} sops=${sopN} contacts=${contactN}`)
      if (failed) { console.error(`${failed} failed`); process.exit(1) }
      console.log('\nSeed probe passed')
      await prisma.$disconnect()
    }

    main().catch(async (err) => {
      console.error(err)
      await prisma.$disconnect()
      process.exit(1)
    })
    ```
